#include <iostream>
#include <cmath>
using namespace std;

bool is_prime(int number) {
    if (number <= 1) {
        return false;
    }
    if (number == 2) {
        return true;
    }
    if (number % 2 == 0) {
        return false;
    }

    int limit = int(sqrt(number));
    for (int i = 3; i <= limit; i += 2) {
        if (number % i == 0) {
            return false;
        }
    }

    return true;
}

int main() {
    int number;
    cout << "Enter a number: ";
    cin >> number;

    if (is_prime(number)) {
        cout << "The number is prime.\n";
    } else {
        cout << "The number is not prime.\n";
    }

    return 0;
}
